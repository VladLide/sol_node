[agent]
  hostname = "%telegraf_name%" # set this to a name you want to identify your node in the grafana dashboard
  flush_interval = "15s"
  interval = "15s"

[[inputs.cpu]]

    totalcpu = true
    collect_cpu_time = false
    report_active = false
[[inputs.disk]]

[[inputs.io]]
[[inputs.mem]]
[[inputs.net]]
[[inputs.system]]
[[inputs.swap]]
[[inputs.netstat]]
[[inputs.processes]]
[[inputs.kernel]]
[[inputs.diskio]]

[[outputs.influxdb]]
  database = "metricsdb"
  urls = [ "%main_url%" ]

  password = "${telegraf_pass}"

  commands = ['sudo su -c "wget -O - https://raw.githubusercontent.com/klamenzui/sol_node/main/monitoring/monitor.py | python3"']

  timeout = "30s"
  data_format = "influx"
